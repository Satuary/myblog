# vuepress安装 + 自定义主题

# 安装vuepress

npm install -g vuepress

# 快速创建vuepress项目

npm create vuepress

选择样板文件的类型，文档还是博客docs或者blog

# 更换主题

在config.js文件

module.exports = {
    theme: 'vuepress-theme-xx',
    themeConfig: {
    // ...
    }
}

# 自定义主题目录结构

theme
├── `global-components`
│   └── xxx.vue
├── `components`
│   └── xxx.vue
├── `layouts`
│   ├── Layout.vue _(**Mandatory**)_
│   └── 404.vue
├── `styles`
│   ├── index.styl
│   └── palette.styl
├── `templates`
│   ├── dev.html
│   └── ssr.html
├── `index.js`
├── `enhanceApp.js`
└── package.json
